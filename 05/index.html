  • быстрый
  • гибкий
  • минималистичный
  • веб-фреймворк
  • для Node.js

Установка


npm install express

Hello World: Node


const http = require('http');

const server = http.createServer((req, res) => {
  res.end('Hello world!');
});

server.listen(3000, '127.0.0.1', () => {
  console.log('Running');
});

Hello World: Express


const express = require('express');
const app = express();

app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(3000, function () {
  console.log('Running');
});

Hello World: Run


node index.js

Статика


app.use(express.static('public'));
        

app.use(express.static('files'));
